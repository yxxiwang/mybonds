Table 1: Graph by columns
col1 col2 col3
1 10 20 30
2 20 40 60
3 40 80 120

$('#table1').graphTable({series: 'columns'});

Table 2: Graph by rows
1 2 3
row1 10 20 30
row2 20 40 60
row3 40 80 120

$('#table2').graphTable({series: 'rows'});

Table 3: Graph by rows, replace currency symbol
1 2 3
row1 $10 $20 $30
row2 $20 $40 $60
row3 $40 $80 $120

$('#table3').graphTable({series: 'rows', dataTransform: function(s) { return(s.replace('$','')); } });
